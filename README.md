# ubuntu 20.04 + ros/noetic

# 安装依赖：
- sudo apt install ros-noetic-rviz-imu-plugin
- sudo apt-get install ros-noetic-ros-control ros-noetic-ros-controllers 
- sudo apt-get install ros-noetic-effort-controllers
